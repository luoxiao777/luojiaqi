{{/* Renders a 2-column image grid for a project Page (.). Image sources: Params.gallery (URLs), else files under Params.imageDir in static/. Images open in site lightbox (a[data-lightbox="1"]). */}} {{- $p := . -}} {{- $imgs := slice -}} {{- with $p.Params.gallery -}} {{- if gt (len .) 0 -}} {{- $imgs = . -}} {{- end -}} {{- end -}} {{- if and (eq (len $imgs) 0) $p.Params.imageDir -}} {{- $dir := printf "static/%s" $p.Params.imageDir -}} {{- range (readDir $dir) -}} {{- $name := .Name -}} {{- $ext := lower (path.Ext $name) -}} {{- if in (slice ".jpg" ".jpeg" ".png" ".webp" ".gif" ".svg") $ext -}} {{- $rel := printf "%s/%s" $p.Params.imageDir $name -}} {{- $u := $rel | relURL -}} {{- $imgs = $imgs | append $u -}} {{- end -}} {{- end -}} {{- end -}}
{{- range $imgs -}}
{{- end -}}
